#######Confim data read in correctly
input <- function(df){
  ### Look at dimensions of dataset
  d<-dim(df)
  ### Identify class of dataset
  c<-class(df)
  ### Look at class of each variable in dataset
  c2<-sapply(df, class)
  ### Bind d, c, and c2 into a list
  ins<-list(d, c, c2)
  ### Return the list of input stats
  return(ins)
}

Identify number of career awards per player

awards <- mutate(awards, awards=1)
awards_num <- awards %>% group_by(playerID) %>% summarize(awards = sum(awards))

Sum Career Stats

players_career <- players %>% group_by(playerID) %>% summarize(year=min(year), 
                                                               GP=sum(GP), GS=sum(GS), minutes=sum(minutes), 
                                                               points=sum(points), oRebounds=sum(oRebounds),
                                                               dRebounds=sum(dRebounds), rebounds=sum(rebounds),
                                                               assists=sum(assists), steals=sum(steals), blocks=sum(blocks), 
                                                               turnovers=sum(turnovers), PF=sum(PF), fgAttempted=sum(fgAttempted),
                                                               fgMade=sum(fgMade), ftAttempted=sum(ftAttempted), ftMade=sum(ftMade),
                                                               threeAttempted=sum(threeAttempted), threeMade=sum(threeMade))

Merge files

data <- merge(players_career, awards_num, by="playerID", all.x=TRUE)
